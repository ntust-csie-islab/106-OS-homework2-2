struct proc*
shmemowner(uint token) {
struct proc owner = (void) 0, *p;

acquire(&ptable.lock);
for(p = ptable.proc; p < &ptable.proc[NPROC]; p++) {
if(p->state == UNUSED)
continue;
if (p->shmem_tok == token) {
owner = p;
break;
}
}
release(&ptable.lock);
return owner;
}

  mappages(proc->pgdir, (char) a, PGSIZE, pa, PTEW|PTEU);
